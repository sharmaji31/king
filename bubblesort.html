#include <stdio.h>
#include <conio.h>
void main()
{
int arr[5]={5,4,3,2,1},i,j,temp;
for(i=0;i<5;i++)
{
for(j=0;j<5;j++)
{
if(arr[j]>arr[j+1])
{
temp=arr[j];
arr[j]=arr[j+1];
arr[j+1]=temp;
}
}
}
printf("your sorted array is:=\n");
for(i=0;i<5;i++)
{
printf("%d",arr[i]);
}
getch();
}
