#include<stdio.h>
#include<conio.h>
#include<time.h>
void main()
{
time_t start=time(NULL);
int arr[5]={5,4,3,2,1},i,j,temp;
for(i=0;i<5;i++)
{
for(j=0;j<5;j++)
{
if(arr[j]>arr[j+1])
{
temp=arr[j];
arr[j]=arr[j+1];
arr[j+1]=temp;
}
}
}
printf("your sorted array is:=\n");
for(i=0;i<5;i++)
{
printf("%d",arr[i]);
}
time_t end=time(NULL);
printf("\nExecution time is %d seconds",(end-start));
getch();
}
