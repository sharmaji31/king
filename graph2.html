#include <stdio.h>l
#include <stdbool.h>
int graph[100][100];
bool visited[100];
int queue[100];
int front = -1, rear = -1;
void enqueue(int node)
{
if (rear == 100- 1)
{
printf("Queue is full.\n");
return;
}
if (front == -1)
{
front = 0;
}
rear++;
queue[rear] = node;
}
int dequeue()
{
if (front == -1)
{
printf("Queue is empty.\n");
return -1;
}
int node = queue[front];
if (front == rear)
{
front = rear = -1;
}
else
{
front++;
}
return node;
}
void breadthFirstTraversal(int start, int numNodes)
{
visited[start] = true;
enqueue(start);
while (front != -1)
{
int currentNode = dequeue();
printf("%d ", currentNode);
for (int i = 0; i < numNodes; i++)
{
if (graph[currentNode][i] == 1 && !visited[i])
{
visited[i] = true;
enqueue(i);
}
}
}
}
int main()
{
int numNodes, numEdges;
printf("Enter the number of nodes and edges: ");
scanf("%d %d", &numNodes, &numEdges);
for (int i = 0; i < numNodes; i++)
{
visited[i] = false;
for (int j = 0; j < numNodes; j++)
{
graph[i][j] = 0;
}
}
printf("Enter the edges (format: source destination):\n");
for (int i = 0; i < numEdges; i++)
{
int source, destination;
scanf("%d %d", &source, &destination);
graph[source][destination] = 1;
}
int startNode;
printf("Enter the starting node: ");
scanf("%d", &startNode);
printf("Breadth-First Traversal starting from node %d: ", startNode);
breadthFirstTraversal(startNode, numNodes);
return 0;
}
