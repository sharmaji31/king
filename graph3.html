#include <stdio.h>
#include <stdbool.h>
int adjacencyMatrix[100][100];
bool visited[100];
int queue[100];
int front = -1, rear = -1;
void enqueue(int node)
{
if (rear == 100 - 1)
{
printf("Queue is full. Cannot enqueue.\n");
return;
}
if (front == -1)
{
front = 0;
}
rear++;
queue[rear] = node;
}
int dequeue()
{
if (front == -1)
{
printf("Queue is empty. Cannot dequeue.\n");
return -1;
}
int node = queue[front];
front++;
if (front > rear)
{
front = rear = -1;
}
return node;
}
void BFS(int start, int nodes)
{
for (int i = 0; i < nodes; i++)
{
visited[i] = false;
}
visited[start] = true;
enqueue(start);
while (front != -1)
{
int currentNode = dequeue();
printf("%d ", currentNode);
for (int i = 0; i < nodes; i++)
{
if (adjacencyMatrix[currentNode][i] && !visited[i])
{
visited[i] = true;
enqueue(i);
}
}
}
}
int main()
{
int nodes;
printf("Enter the number of nodes: ");
scanf("%d", &nodes);
printf("Enter the adjacency matrix:\n");
for (int i = 0; i < nodes; i++)
{
for (int j = 0; j < nodes; j++)
{
scanf("%d", &adjacencyMatrix[i][j]);
}
}
int startNode;
printf("Enter the starting node: ");
scanf("%d", &startNode);
printf("Breadth-First Search starting from node %d: ", startNode);
BFS(startNode, nodes);
return 0;
}
