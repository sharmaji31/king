int a[11] = { 10, 14, 19, 26, 27, 31, 33, 35, 42, 44};
int b[10];
int main()
{
int i,low=0,high=9;
sort(low,high);
merging(low,high);
printf("\nList after sorting\n");
for(i = 0; i <10; i++)
printf("%d ", a[i]);
}
int sort(int low, int high)
{
int mid;
while(low {
mid =(low + high) / 2;
high=mid;
}
}
int merging(int low, int mid, int high) { int l1, l2, i;
for(l1 = low, l2 = mid + 1, i = low; l1 <= mid && l2 <= high; i++)
{ if(a[l1] <= a[l2])
{
b[i] = a[l1];
l1++;
}
else
{
b[i] = a[l2];
l2++;
}
}
while(l1 <= mid)
{
b[i] = a[l1];
i++;l1++;
}
while(l2 <= high)
{
b[i] = a[l2];
i++;l2++;
}
for(i = low; i <= high; i++)
a[i] = b[i];
}
