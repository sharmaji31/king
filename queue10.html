#include <conio.h>
#include <stdlib.h>
struct node
{
int data;
int p;
struct node *link;
};
struct node *pointer;
struct node *start;
struct node *new,*temp;
void main()
{
int n=1,a;
while(n!=0)
{
printf("\npress the button=");
scanf("%d",&a);
if(a<=3) { switch(a)
{
case 1:push();
break;
case 2:pop();
break; case 3:return 0; break; }
}
else { printf("press valid button\n"); } } return 0;
}
int push()
{
if(start==NULL)
{
start=malloc(sizeof(struct node));
printf("\nenter the value=");
scanf("%d",&start->data);
printf("\nenter the priority=");
scanf("%d",&start->p);
start->link=NULL;
pointer=start;
temp=start;
}
else
{
new=malloc(sizeof(struct node));
printf("\nenter the value=");
scanf("%d",&new->data);
printf("\nenter the
priority=");
scanf("%d",&new->p);
new->link=NULL;
if(pointer->p>new->p)
{
temp->link=new;
}
else
{
new->link=pointer;
pointer=new;
}
}
}
int pop()
{
printf("value according to priority\n");
while(pointer!=NULL)
{
printf("%d ",pointer->data);
pointer=pointer->link;
}
}
