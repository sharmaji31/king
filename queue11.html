#include <stdio.h>
#include <conio.h>
#include <stdlib.h>
struct node
{
char data[100];
int p;
struct node *link;
};
struct node *pointer;
struct node *start;
struct node *new,*temp;
void main()
{
int n=1,a;
while(n!=0)
{
printf("\npress the button=");
scanf("%d",&a);
switch(a)
{
case 1:push();
break;
case 2:pop();
break;
}
}
getch();
}
int push()
{
if(start==NULL)
{
start=malloc(sizeof(struct node));
printf("\nenter the string=");
scanf("%s",&start->data);
printf("\nenter the priority=");
scanf("%s",&start->p);
start->link=NULL;
pointer=start;
temp=start;
}
else
{
new=malloc(sizeof(struct node));
printf("\nenter the string=");
scanf("%s",&new->data);
printf("\nenter the priority=");
scanf("%s",&new->p);
new->link=NULL;
if(pointer->p>new->p)
{
temp->link=new;
}
else
{
new->link=pointer;
pointer=new;
}
}
}
int pop()
{
printf("value according to priority\n");
while(pointer!=NULL)
{
printf("%s ",pointer->data);
pointer=pointer->link;
}
}

OUTPUT:
Press1 for push
Press2 for pop

Enter the priority=1
