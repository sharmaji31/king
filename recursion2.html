#include <stdio.h>
#include <conio.h>
int binary(int,int,int,int);
void main()
{
int n,rem,sum=0,base=1;
printf("enter the decimal number=");
scanf("%d",&n);
sum=binary(n,rem,sum,base);
printf("After converting decimal to binary=%d",sum);
getch();
}
int binary(int n,int rem,int sum,int base)
{
if(n>0)
{
rem=n%2;
sum=sum+rem*base;
n=n/2;
base=base*10;
return binary(n,rem,sum,base);
}
return sum;
}

OUTPUT:
Enter the decimal number=9
After converting number from decimal to binary=1001
