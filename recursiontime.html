#include<stdio.h>
#include<conio.h>
#include<time.h>
int binary(int,int,int,int);
void main()
{
time_t start=time(NULL);
int n,rem,sum=0,base=1;
printf("enter the binary number=");
scanf("%d",&n);
sum=binary(n,rem,sum,base);
printf("After converting binary to decimal=%d",sum);
time_t end=time(NULL);
printf("\nExecution time is %d seconds",(end-start));
getch();
}
int binary(int n,int rem,int sum,int base)
{
if(n>0)
{
rem=n%10;
sum=sum+rem*base;
n=n/10;
base=base*2;
return binary(n,rem,sum,base);
}
return sum;
}
