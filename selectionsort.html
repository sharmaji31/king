#include <stdio.h>
#include <conio.h>
void main()
{
int arr[5]={5,4,3,2,1},i,j,temp;
for(i=0;i<5-1;i++)
{
for(j=i+1;j<5;j++)
{
if(arr[i]>arr[j])
{
temp=arr[i];
arr[i]=arr[j];
arr[j]=temp;
}
}
}
printf("your sorted array is:=\n");
for(i=0;i<5;i++)
{
printf("%d",arr[i]);
}
getch();
}
