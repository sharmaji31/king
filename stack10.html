#include <stdio.h>
#include <conio.h>
void main()
{
char str[100];
int arr[10];
int top=0,i=0,count=0;
printf("enter the string=");
gets(str);
for(i=0;str[i];i++)
{
count++;
if(str[i]!=' ')
{
arr[top]=str[i];
top++;
}
}
printf("After reversing string=\n");
for(top=count;top>=0;top--)
{
printf("%c",arr[top]);
}
getch();
}
