int vowel(char [],int,int);
void main()
{
char str[100];
int i,count=0;
printf("enter the word=");

count=vowel(str,i,count);
printf("total no of vowel=%d",count);

}
int vowel(char str[],int i,int count)
{
for(i=0;str[i];i++)
{
switch(str[i])
{
case 'a':
case 'i':
case 'o':
case 'u':
count++;
}
}
return count;
}
