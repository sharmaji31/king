#include <stdio.h>
#include <stdlib.h>
struct TreeNode
{
int data;
struct TreeNode* left;
struct TreeNode* right;
};
struct TreeNode* createNode(int value)
{
struct TreeNode* newNode = (struct TreeNode*)malloc(sizeof(struct TreeNode));
newNode->data = value;
newNode->left = newNode->right = NULL;
return newNode;
}
struct TreeNode* insertNode(struct TreeNode* root, int value)
{
if (root == NULL)
{
return createNode(value);
}
if (value <= root->data)
{
root->left = insertNode(root->left, value);
}
else
{
root->right = insertNode(root->right, value);
}
return root;
}
struct TreeNode* createCompleteBinaryTree()
{
struct TreeNode* root = NULL;
int value;
printf("Enter elements to insert into the binary tree (-1 to stop):\n");
while (1)
{
scanf("%d", &value);
if (value == -1)
{
break;
}
root = insertNode(root, value);
}
return root;
}
void inorderTraversal(struct TreeNode* root)
{
if (root == NULL) return;
inorderTraversal(root->left);
printf("%d ", root->data);
inorderTraversal(root->right);
}
void freeTree(struct TreeNode* root)
{
if (root == NULL) return;
freeTree(root->left);
freeTree(root->right);
free(root);
}
int main()
{
struct TreeNode* root = createCompleteBinaryTree();
printf("Inorder traversal of the complete binary tree: ");
inorderTraversal(root);
printf("\n");
freeTree(root);
return 0;
}
